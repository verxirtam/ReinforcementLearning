Messageのフォーマット
行末の\t*;.*はコメント。使用時は在ってはならない
//メッセージ全体/////////////////////////////////////////////
//
//
/////////////////////////////////////////////////////
MESSAGE_1.0			;メッセージ開始
<MessageBody>		;メッセージの本体
					;(空行)メッセージ終了
////////////////////////////////////////////////////////
<MessageBody>:
	<Version>			;バージョン
	<Command>|<Result>	;コマンド|コマンドの結果
	<Body>				;コマンドの内容
////////////////////////////////////////////////////////
<Version>:
	EV3LineTracer_1.0	;バージョン
////////////////////////////////////////////////////////
<Command>:			;コマンド
	SetMDP			;MDPを設定
	|SetPolicy		;CurrentPolicyを設定
	|ExecEpisode	;Episodeを実行
////////////////////////////////////////////////////////
<Result>:			;結果
	OK				;コマンド実行に成功
	|NG				;コマンド実行に失敗
////////////////////////////////////////////////////////
<Body>:
	(任意の文字列(ただし空行(\n\n)を含んではならない))	;詳細は各コマンドの詳細のとおり
////////////////////////////////////////////////////////
