
<Message>::=<MessageVersion><endl> ;メッセージのバージョン
				<MessageBody>;メッセージ本体
				<endl>;空行

<MessageVersion>::="MESSAGE_1.0"

<MessageBody>::=<MessageBodyVersion><endl>;バージョン
						<Command>|<Result>;コマンド|コマンドの結果

<Command>::=
(
	"SetMDP"<endl>;MDPを設定
	(
		<SetMDPBody>;コマンドの内容
		|
		<SetMDPResult>;コマンドの結果
	)
	|
	"SetPolicy"<endl>;MDPを設定
	(
		<SetPolicyBody>;コマンドの内容
		|
		<SetPolicyResult>;コマンドの結果
	)
	|
	"ExecEpisode"<endl>;Episodeを実行
	(
		<Null>;コマンドの内容(無し)
		|
		<ExecEpisodeResult>;コマンドの結果
	)
	|
	"NullCommand"<endl>;何もしないコマンド
	(
		<Null>;コマンドの内容(無し)
		|
		<NullCommandResult>;コマンドの結果
	)
)					

<Null>::="";空文字列


<SetMDPBody>::=
	<Interval><endl>;Interval(msec)
	<StateCount><endl>;State数
	<State>;Stateの定義
	<Control>;Controlの定義(M=全てのStateの全てのControl数の合計)
	<RegularPolicy>;RegularPolicyの定義

<State>::=
	N(<SingleState><endl>);Stateの定義 (StateIndexは0からN-1まで順に並ぶ(N=StateCount))

<SingleState>::=
	<StateIndex><tab>;StateIndex
	<RefMax><tab>;RefMax
	<ControlCount>;ControlCount

<StateIndex>::=DIGIT
<RefMax>::=DIGIT"."DIGIT
<ControlCount>::=DIGIT

<Control>::=
	M(<SingleControl><endl>)
		;(Controlの定義 StateIndexは0からN-1まで順に並ぶ(N=StateCount)
		   ControlIndexは０からStateのControlCount-1まで順に並ぶ)

<SingleControl>::=
	<StateIndex><tab>		;StateIndex
	<ControlIndex><tab>	;ControlIndex
	<LMotorSpeed><tab>	;LMotorSpeed
	<RMotorSpeed>			;RMotorSpeed

<ControlIndex>::=DIGIT
<LMotorSpeed>::=DIGIT
<RMotorSpeed>::=DIGIT

<RegularPolicy>::=
	N(<SinglePolicy><endl>);RegularPolicyの定義(StateIndexは0からN-1まで順に並ぶ(N=StateCount))

<SinglePolicy>::=
	<StateIndex><tab>;StateIndex
	<ControlIndex>;ControlIndex

<SetMDPResult>::=
	(
		"OK"<endl>
		|
		(
			"NG"<endl>
			<Reason>
		)
	)

<ExecEpisodeResult>::=
(
	"OK"<endl>
	<EpisodeBody>
	|
	"NG"<endl>
	<Reason>
)

<EpisodeBody>::=
	<StepCount><endl>
	N(<SingleStep><endl>)	;NはStepCountの値

<StepCount>::=DIGIT

<SingleStep>::=
	<EpisodeIndex><tab>
	<State><tab>
	<Control><tab>
	<Cost>

<EpisodeIndex>::=DIGIT
<State>::=DIGIT
<Control>::=DIGIT
<Cost>::=DIGIT"."DIGIT


<Reason>::=(発生した例外のStackTrace)



<NullCommandResult>="OK"

Messageのフォーマット
行末の\t*;.*はコメント。使用時は在ってはならない
//メッセージ全体/////////////////////////////////////////////
//
//
/////////////////////////////////////////////////////
MESSAGE_1.0			;メッセージ開始
<MessageBody>		;メッセージの本体
					;(空行)メッセージ終了
////////////////////////////////////////////////////////
<MessageBody>:
	<Version>			;バージョン
	<Command>|<Result>	;コマンド|コマンドの結果
////////////////////////////////////////////////////////
<Version>:
	EV3LineTracer_1.0	;バージョン
////////////////////////////////////////////////////////
<Command>:
	<CommandCategory>;コマンド種別
	<Body>				;コマンドの内容
////////////////////////////////////////////////////////
<CommandCategory>:			;コマンド種別
	SetMDP			;MDPを設定
	|SetPolicy		;CurrentPolicyを設定
	|ExecEpisode	;Episodeを実行
////////////////////////////////////////////////////////
<Result>:			;結果
	OK				;コマンド実行に成功
	|NG				;コマンド実行に失敗
////////////////////////////////////////////////////////
<Body>:
	(任意の文字列(ただし空行(\n\n)を含んではならない))	;詳細は各コマンドの詳細のとおり
////////////////////////////////////////////////////////
